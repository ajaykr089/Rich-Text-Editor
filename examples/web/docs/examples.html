React Example

import React, { useState } from 'react';
import EditoraEditor from '@editora/react';

export default function Editor() {
  const [content, setContent] = useState('<p>Start typing...</p>');

  const handleChange = (newContent) => {
    setContent(newContent);
    console.log('Content:', newContent);
  };

  return (
    <div>
      <h1>My Editor</h1>
      <EditoraEditor
        defaultContent={content}
        onChange={handleChange}
      />
      <button onClick={() => console.log('Saved:', content)}>
        Save
      </button>
    </div>
  );
}

Angular Example

// app.component.ts
import { Component, AfterViewInit, OnDestroy } from '@angular/core';
import { Editor } from '@editora/core';

@Component({
  selector: 'app-root',
  template: '<div id="editor"></div>'
})
export class AppComponent implements AfterViewInit, OnDestroy {
  private editor?: Editor;

  ngAfterViewInit() {
    this.editor = new Editor({
      element: document.getElementById('editor') as HTMLElement,
      content: '<p>Hello from Angular!</p>'
    });
  }

  ngOnDestroy() {
    this.editor?.destroy?.();
  }
}

With Plugins Example

import { Editor } from '@editora/core';
import { MediaManagerPlugin } from '@editora/plugins/media-manager';
import { TablePlugin } from '@editora/plugins/table';
import { CodeSamplePlugin } from '@editora/plugins/code-sample';

const editor = new Editor({
  element: '#editor',
  content: '<p>Start editing...</p>',
  plugins: [
    new MediaManagerPlugin({
      uploadUrl: '/api/upload'
    }),
    new TablePlugin(),
    new CodeSamplePlugin()
  ]
});

// Save content
document.getElementById('save').addEventListener('click', () => {
  const html = editor.getContent();
  // Send to server
  fetch('/api/save', {
    method: 'POST',
    body: JSON.stringify({ content: html })
  });
});

Auto-Save Example

import { Editor } from '@editora/core';

const editor = new Editor({
  element: '#editor'
});

// Auto-save every 30 seconds
setInterval(() => {
  const content = editor.getContent();
  localStorage.setItem('draft', content);
  console.log('Auto-saved');
}, 30000);

// Load draft on page load
window.addEventListener('DOMContentLoaded', () => {
  const draft = localStorage.getItem('draft');
  if (draft) {
    editor.setContent(draft);
  }
});

Event Handling Example

const editor = new Editor({
  element: '#editor'
});

editor.on('change', (content) => {
  console.log('Content changed:', content);
  // Update preview
  document.getElementById('preview').innerHTML = content;
});

editor.on('focus', () => {
  console.log('Editor focused');
  document.getElementById('status').textContent = 'Editing...';
});

editor.on('blur', () => {
  console.log('Editor blurred');
  document.getElementById('status').textContent = 'Ready';
});

editor.on('selectionChange', (selection) => {
  if (selection.isBold) {
    document.getElementById('bold-btn').classList.add('active');
  }
});

Data Binding Example

// Bind to form
const form = document.getElementById('form');
const contentInput = document.getElementById('content');

const editor = new Editor({
  element: '#editor'
});

// Set initial content
editor.setContent(contentInput.value);

// Update input on change
editor.on('change', (content) => {
  contentInput.value = content;
});

// Submit form
form.addEventListener('submit', (e) => {
  e.preventDefault();
  // Form is ready with content in contentInput
  form.submit();
});

Formatting Example

const editor = new Editor({
  element: '#editor'
});

// Make selected text bold
function makeBold() {
  editor.execCommand('bold');
}

// Insert heading
function insertHeading(level = 1) {
  editor.execCommand('heading', { level });
}

// Insert link
function insertLink() {
  const url = prompt('Enter URL:');
  if (url) {
    editor.execCommand('insertLink', { url });
  }
}

// Clear formatting
function clearFormatting() {
  editor.execCommand('clearFormatting');
}

Validation Example

const editor = new Editor({
  element: '#editor'
});

function validateContent() {
  const content = editor.getContent();
  
  // Check if empty
  if (content === '<p></p>' || content.trim() === '') {
    alert('Content cannot be empty');
    return false;
  }
  
  // Check minimum length
  const text = new DOMParser().parseFromString(content, 'text/html').body.textContent;
  if (text.length < 10) {
    alert('Content must be at least 10 characters');
    return false;
  }
  
  return true;
}

document.getElementById('submit').addEventListener('click', () => {
  if (validateContent()) {
    // Save content
    saveContent(editor.getContent());
  }
});
